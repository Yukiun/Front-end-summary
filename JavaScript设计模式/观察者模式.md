# 一个简单的观察者模式例子
```js
class Observer {
    constructor() {
        this.listener = new Map();
    }
    addEventListener(event, cb) {
        const cbs = this.listener.get(event) || [];
        cbs.push(cb);
        this.listener.set(event, cbs);
    }
    emit(event) {
        const cbs = this.listener.get(event) || [];
        cbs.forEach(cb => cb());
    }
}

const ob = new Observer();

ob.addEventListener(1, () => {
    console.log(111111);
});

ob.addEventListener(1, () => {
    console.log('呵呵');
});

ob.addEventListener(22, () => {
    console.log(22222);
})


for (let i = 0; i < 5; i++) {
    ob.emit(i);
}
```
